#!/usr/bin/php
<?php

%%COMMAND_USE%% as AppCommand;
use Mediatag\Core\MediaApplication;

define('__ROOT_DIRECTORY__', dirname(realpath($_SERVER['SCRIPT_FILENAME']), 2));
define('__CONFIG_LIB__', __ROOT_DIRECTORY__.'/config');
$app         = require __ROOT_DIRECTORY__.'/bootstrap.php';
$command     = new AppCommand();
$application = new MediaApplication($command->getName(), '1.0.0');
$application->add($command);
$application->setDefaultCommand($command->getName(), true);
$application->run();
